# 1. 敏感词过滤方案

- 正则匹配替换：正则性能较差，词库过大，性能低
- Trie树：字典树，又称单词查找树。它是哈希树的变种。（前缀数）



# 2. 实现

```go
type Node struct {
	char     rune
	Data     interface{}
	parent   *Node
	Depth    int
	children map[rune]*Node
	term     bool
}

type Trie struct {
	root *Node
	size int
}

func NewNode() *Node {
	return &Node{
		children: make(map[rune]*Node, 32),
	}
}

func NewTrie() *Trie {
	return &Trie{
		root: NewNode(),
	}
}

func (p *Trie) Add(key string, data interface{}) (err error) {
	node := p.root
	for _, c := range key {
		ret, ok := node.children[c]
		if !ok {
			ret = NewNode()
			ret.Depth = node.Depth + 1
			ret.char = c
			node.children[c] = ret
		}

		node = ret
	}

	node.term = true
	node.Data = data
	return
}

func (p *Trie) findNode(key string) (result *Node) {
	node := p.root
	for _, v := range key {
		ret, ok := node.children[v]
		if !ok {
			return
		}

		node = ret
	}

	result = node
	return
}

func (p *Trie) collectNode(node *Node) (result []*Node) {
	if node == nil {
		return
	}

	if node.term {
		result = append(result, node)
		return
	}

	var queue []*Node
	queue = append(queue, node)

	for i := 0; i < len(queue); i++ {
		if queue[i].term {
			result = append(result, queue[i])
			continue
		}

		for _, v := range queue[i].children {
			queue = append(queue, v)
		}
	}

	return
}

func (p *Trie) PrefixSearch(key string) (result []*Node) {
	node := p.findNode(key)
	if node == nil {
		return
	}

	result = p.collectNode(node)
	return
}

func (p *Trie) Check(text, replace string) (result bool, str string) {
	node := p.root
	if node == nil {
		return
	}

	chars := []rune(text)

	var left []rune
	var start int
	for i, v := range chars {
		ret, ok := node.children[v]
		if !ok {
			left = append(left, chars[start:i+1]...)
			start = i + 1
			node = p.root
			continue
		}

		node = ret
		if ret.term {
			result = true
			node = p.root
			left = append(left, []rune(replace)...)
			start = i + 1
			continue
		}
	}

	str = string(left)
	return
}
```

func TestTrie(t *testing.T) {

	trie.Add("白痴", nil)

	ret, str := trie.Check("你是个白痴，fuck U!", "***")
